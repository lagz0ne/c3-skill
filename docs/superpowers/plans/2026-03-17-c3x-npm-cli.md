# @c3x/cli npm Package — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a thin npm CLI (`@c3x/cli`) that discovers installed c3x Go binaries and delegates to them, giving humans a `npx @c3x/cli <cmd>` entry point.

**Architecture:** Node.js thin client written in TypeScript, built with tsdown to a single ESM file. Discovers c3x installations across project/claude/codex/marketplace paths, picks highest version, delegates via `c3x.sh`. Go binary gains `C3X_MODE` env var for human/agent output switching.

**Tech Stack:** TypeScript, tsdown (Rolldown), Node.js stdlib only (zero runtime deps), Go (for C3X_MODE support)

**Spec:** `docs/superpowers/specs/2026-03-17-c3x-npm-cli-design.md`

### Task 2: Write CLI source — discovery + delegation

**Files:**
- Create: `packages/cli/src/cli.ts`

The source has 3 concerns:
1. **Arg parsing** — extract `--agent` flag, pass rest through
2. **Discovery** — scan paths, read VERSION files, pick highest semver
3. **Delegation** — spawn `c3x.sh` with inherited stdio/cwd

- [ ] **Step 1: Write `src/cli.ts`**

```typescript
import { execFileSync } from 'node:child_process'
import { existsSync, readFileSync, readdirSync, statSync } from 'node:fs'
import { dirname, join, resolve } from 'node:path'
import { homedir } from 'node:os'

// --- Arg parsing ---

interface ParsedArgs {
  agent: string | undefined  // "claude" | "codex" | undefined (all)
  rest: string[]             // args forwarded to c3x
}

function parseArgs(argv: string[]): ParsedArgs {
  const rest: string[] = []
  let agent: string | undefined

  for (let i = 0; i < argv.length; i++) {
    if (argv[i] === '--agent' && i + 1 < argv.length) {
      agent = argv[++i]
    } else {
      rest.push(argv[i])
    }
  }
  return { agent, rest }
}

// --- Version comparison ---

function parseSemver(v: string): [number, number, number] | null {
  const m = v.trim().match(/^(\d+)\.(\d+)\.(\d+)$/)
  if (!m) return null
  return [Number(m[1]), Number(m[2]), Number(m[3])]
}

function compareSemver(a: [number, number, number], b: [number, number, number]): number {
  for (let i = 0; i < 3; i++) {
    if (a[i] !== b[i]) return a[i] - b[i]
  }
  return 0
}

// --- Discovery ---

interface Candidate {
  binDir: string    // path to skills/c3/bin/
  version: [number, number, number]
  priority: number  // lower = higher priority (for tie-breaking)
}

function readVersion(binDir: string): [number, number, number] | null {
  const versionFile = join(binDir, 'VERSION')
  try {
    if (!existsSync(versionFile)) return null
    const raw = readFileSync(versionFile, 'utf-8')
    return parseSemver(raw)
  } catch {
    return null
  }
}

function addCandidate(candidates: Candidate[], binDir: string, priority: number): void {
  const version = readVersion(binDir)
  if (version) candidates.push({ binDir, version, priority })
}

function discoverProjectScope(cwd: string): string[] {
  const results: string[] = []
  let dir = cwd
  while (true) {
    const candidate = join(dir, 'skills', 'c3', 'bin')
    if (existsSync(join(candidate, 'VERSION'))) {
      results.push(candidate)
    }
    // Stop at .git root
    if (existsSync(join(dir, '.git'))) break
    const parent = dirname(dir)
    if (parent === dir) break // filesystem root
    dir = parent
  }
  return results
}

function discoverMarketplace(): string[] {
  const base = join(homedir(), '.claude', 'plugins', 'marketplaces')
  if (!existsSync(base)) return []
  const results: string[] = []
  try {
    for (const entry of readdirSync(base)) {
      const candidate = join(base, entry, 'skills', 'c3', 'bin')
      if (existsSync(join(candidate, 'VERSION'))) {
        results.push(candidate)
      }
    }
  } catch { /* skip */ }
  return results
}

function discover(cwd: string, agentFilter: string | undefined): Candidate[] {
  const candidates: Candidate[] = []
  let priority = 0

  // 1. Project scope (always included)
  for (const binDir of discoverProjectScope(cwd)) {
    addCandidate(candidates, binDir, priority++)
  }

  // 2. Claude skills
  if (!agentFilter || agentFilter === 'claude') {
    const claudeSkills = join(homedir(), '.claude', 'skills', 'c3', 'bin')
    addCandidate(candidates, claudeSkills, priority++)
  }

  // 3. Codex skills
  if (!agentFilter || agentFilter === 'codex') {
    const codexSkills = join(homedir(), '.codex', 'skills', 'c3', 'bin')
    addCandidate(candidates, codexSkills, priority++)
  }

  // 4. Marketplace (claude-only scope)
  if (!agentFilter || agentFilter === 'claude') {
    for (const binDir of discoverMarketplace()) {
      addCandidate(candidates, binDir, priority++)
    }
  }

  return candidates
}

function pickBest(candidates: Candidate[]): Candidate | null {
  if (candidates.length === 0) return null
  return candidates.sort((a, b) => {
    const versionDiff = compareSemver(b.version, a.version) // descending
    if (versionDiff !== 0) return versionDiff
    return a.priority - b.priority // ascending (lower = better)
  })[0]
}

// --- Main ---

const { agent, rest } = parseArgs(process.argv.slice(2))
const candidates = discover(process.cwd(), agent)
const best = pickBest(candidates)

if (!best) {
  console.error('error: No c3x installation found.')
  console.error('')
  console.error('Install via Claude Code skill marketplace or Codex:')
  console.error('  claude skill install c3-skill')
  console.error('')
  console.error('Searched:')
  if (!agent || agent === 'claude') {
    console.error('  ~/.claude/skills/c3/bin/')
    console.error('  ~/.claude/plugins/marketplaces/*/skills/c3/bin/')
  }
  if (!agent || agent === 'codex') {
    console.error('  ~/.codex/skills/c3/bin/')
  }
  console.error('  <project>/skills/c3/bin/ (walking up from cwd)')
  process.exit(1)
}

const c3xSh = join(best.binDir, 'c3x.sh')

try {
  execFileSync('bash', [c3xSh, ...rest], {
    stdio: 'inherit',
    cwd: process.cwd(),
  })
} catch (err: any) {
  process.exit(err.status ?? 1)
}
```
